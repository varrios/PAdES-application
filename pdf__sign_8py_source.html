28def decrypt_private_key(private_key_filepath: str, pin: str) -> RSA.RsaKey:
29 try:

48def sign_pdf_file(decrypted_private_key: RSA.RsaKey, pdf_filepath: str) -> None:
49 reader = PdfReader(pdf_filepath)
50 writer = PdfWriter()
