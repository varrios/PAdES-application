87 usb_found_status = ""

107 # If USB detected, search for private key

109 self.usb_path = None
110 self.private_key_found = False
111 self.private_key_path = None
112 usb_found_status += "❌ No USB detected"
113 else:
114 self.usb_path = drives[0]['device']
115 usb_found_status += \
116 (
117 f"🟢 USB detected:\n"
118 f"{drives[0]['device']}{drives[0]['name']}"
119 )
120 # Search for private key
121 private_keys_found = search_usb_for_private_key(usb_path=self.usb_path)
122 if private_keys_found:
123 self.private_key_found = True
124 self.private_key_path = private_keys_found[0]
125 usb_private_key_found_status += \
126 (
127 f"🔑 Private key found on USB at:\n"
128 f"{str(private_keys_found[0])}"
129 )
130 else:
131 self.private_key_found = False
132 usb_private_key_found_status += "❌ No private key found on USB drive"
133
134

137 f"{usb_private_key_found_status}\n"
