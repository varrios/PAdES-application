24def check_for_usb_device() -> tuple[bool, list | None]:
25 usb_disks = []
26
27 if os.name == "nt":
28 for partition in psutil.disk_partitions(all=True):
29 device = partition.device
30
31 drive_type = win32file.GetDriveType(device)
32 if drive_type == win32file.DRIVE_REMOVABLE:
33 try:
34 volume_name, _, _, _, _ = win32api.GetVolumeInformation(device)
35 except Exception:
36 volume_name = "Unknown"
37
38 usb_disks.append({
39 "device": device,
40 "name": volume_name
41 })
42 else: # TESTING PURPOSES ONLY FOR UNIX-BASED SYSTEMS
43 load_dotenv()
44 usb_disks.append({
45 "device": os.getenv('CUSTOM_FILEPATH'),
46 "name": "USB SIMULATON"
47 })
48 if usb_disks:
49 logger.info(f"USB storage devices detected: {len(usb_disks)}")
50 return True, usb_disks
51 else:
52 logger.info("No USB storage devices detected")
